%   Define the longlisting environment
% Breakable, non-float "longlisting"
\newenvironment{longlisting}{%
  % make \caption work outside floats and tighten caption gap
  \captionsetup{type=listing,position=bottom,skip=0pt}%
  % tighten the verbatim block; ensure it may break across pages
  \setminted{
    fontsize=\scriptsize,
    baselinestretch=1,
    breaklines,
    samepage=false
  }%
  % local caption spacing
  \begingroup
  \setlength{\abovecaptionskip}{0pt}%
  \setlength{\belowcaptionskip}{4pt}%
}{%
  \endgroup
}
